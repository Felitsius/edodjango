# My standard file structure for django projects
1. \apps
    - myapp
2. \media
3. \mysite (name project django)
    - \settings
        - __init__.py
        - base.py
        - development.py
        - production.py
    - __init__.py
    - asgi.py
    - urls.py
    - wsgi.py
4. \requirements
    - base.txt
    - development.txt
    - production.txt
5. \static
    - \css
    - \images
    - \js
6. \templates
7. .env
8. .gitignore
9. README.md
10. manage.py
